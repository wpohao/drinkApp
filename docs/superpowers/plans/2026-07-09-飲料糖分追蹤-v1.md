# 飲料糖分追蹤 App V1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 打造一支 iOS App，讓使用者記錄手搖／飲料，自動算糖分與熱量、看今日剩餘額度與歷史趨勢，並用規則式通知提醒，靠 AdMob 變現。

**Architecture:** 模組化 MVVM。`Core` 層（GRDB 資料庫、Model、糖分計算、Repository）與 UI 無關、可獨立單元測試；`Features` 層每個畫面一組 ViewModel＋View；`Shared` 放共用 UI、建議規則、通知、廣告。V1 對外只露飲料糖分功能，但架構保留未來掛入健身／情緒模組的空間。ViewModel 依賴 Repository 的 **protocol**，測試時注入假物件。

**Tech Stack:** Swift 5.9+、SwiftUI、GRDB.swift（SQLite）、Swift Charts（統計圖）、Google Mobile Ads SDK（AdMob）、UserNotifications（本地通知）、XCTest（測試）、Swift Package Manager（相依管理）。

## Global Constraints

- **開發環境**：必須在 **macOS ＋ Xcode 15+** 上開發、build、測試、上架（Windows 無法）。
- **最低 iOS 版本**：iOS 16.0（Swift Charts 需要）。
- **相依管理**：一律用 Swift Package Manager，不用 CocoaPods（避免 `Pods/` 撐大 repo）。`Package.resolved` 要 commit。
- **無 Server、無 AI**：V1 所有資料存本機 SQLite；不呼叫任何雲端 API。
- **數值定位**：糖分／熱量一律為「參考估算」，UI 需可見此標示；不得作醫療宣稱。
- **糖度乘數**：無糖 0、一分糖 0.1、微糖(三分) 0.3、半糖(五分) 0.5、少糖(七分) 0.7、八分糖 0.8、全糖 1.0。
- **容量**：中杯(m) 500ml、大杯(l) 700ml、特大杯(xl) 1000ml；品項 `base_size_ml` 預設 700。
- **糖熱量係數**：糖每公克 4 kcal。
- **每日糖分目標預設**：50g（可改 25g／自訂）。
- **加料**：固定份量，不隨糖度或容量縮放，直接加總。
- **commit 訊息**：遵循團隊規範 `<type>(<scope>): <subject>`（type：feat/fix/docs/chore/test…）。
- **spec 出處**：`docs/superpowers/specs/2026-07-09-飲料糖分追蹤-design.md`（有疑問回查此檔）。

```swift
import Foundation
import GRDB

struct AppDatabase {
    let dbQueue: DatabaseQueue

    init(_ dbQueue: DatabaseQueue) throws {
        self.dbQueue = dbQueue
        try Self.migrator.migrate(dbQueue)
    }

    /// 正式 App 使用：DB 檔放 Application Support/DrinkSugar/db.sqlite
    static func makeShared() throws -> AppDatabase {
        let fm = FileManager.default
        let dir = try fm.url(for: .applicationSupportDirectory, in: .userDomainMask,
                             appropriateFor: nil, create: true)
            .appendingPathComponent("DrinkSugar", isDirectory: true)
        try fm.createDirectory(at: dir, withIntermediateDirectories: true)
        let dbURL = dir.appendingPathComponent("db.sqlite")
        return try AppDatabase(try DatabaseQueue(path: dbURL.path))
    }

    static var migrator: DatabaseMigrator {
        var migrator = DatabaseMigrator()
        migrator.registerMigration("v1_schema") { db in
            try db.create(table: "drink_shop") { t in
                t.autoIncrementedPrimaryKey("id")
                t.column("name", .text).notNull()
                t.column("sort_order", .integer).notNull().defaults(to: 0)
            }
            try db.create(table: "drink_item") { t in
                t.autoIncrementedPrimaryKey("id")
                t.column("shop_id", .integer).notNull().references("drink_shop", onDelete: .cascade)
                t.column("name", .text).notNull()
                t.column("category", .text)
                t.column("base_sugar_g", .double).notNull()
                t.column("base_kcal", .double).notNull()
                t.column("base_size_ml", .integer).notNull().defaults(to: 700)
                t.column("is_estimated", .boolean).notNull().defaults(to: true)
            }
            try db.create(table: "sugar_level") { t in
                t.primaryKey("code", .text)
                t.column("label", .text).notNull()
                t.column("multiplier", .double).notNull()
                t.column("sort_order", .integer).notNull().defaults(to: 0)
            }
            try db.create(table: "size_option") { t in
                t.primaryKey("code", .text)
                t.column("label", .text).notNull()
                t.column("ml", .integer).notNull()
            }
            try db.create(table: "topping") { t in
                t.autoIncrementedPrimaryKey("id")
                t.column("name", .text).notNull()
                t.column("sugar_g", .double).notNull()
                t.column("kcal", .double)
                t.column("is_estimated", .boolean).notNull().defaults(to: true)
                t.column("sort_order", .integer).notNull().defaults(to: 0)
            }
            try db.create(table: "intake_log") { t in
                t.autoIncrementedPrimaryKey("id")
                t.column("logged_at", .datetime).notNull()
                t.column("item_id", .integer).references("drink_item", onDelete: .setNull)
                t.column("custom_name", .text)
                t.column("sugar_g", .double).notNull()
                t.column("kcal", .double)
                t.column("sugar_level_code", .text).references("sugar_level")
                t.column("size_code", .text).references("size_option")
                t.column("note", .text)
            }
            try db.create(table: "intake_log_topping") { t in
                t.autoIncrementedPrimaryKey("id")
                t.column("intake_log_id", .integer).notNull().references("intake_log", onDelete: .cascade)
                t.column("topping_id", .integer).references("topping", onDelete: .setNull)
                t.column("custom_name", .text)
                t.column("qty", .integer).notNull().defaults(to: 1)
                t.column("sugar_g", .double).notNull()
                t.column("kcal", .double)
            }
            try db.create(table: "user_setting") { t in
                t.primaryKey("id", .integer)
                t.column("daily_sugar_target_g", .double).notNull().defaults(to: 50)
                t.column("notify_threshold", .boolean).notNull().defaults(to: true)
                t.column("notify_daily_log", .boolean).notNull().defaults(to: false)
                t.column("notify_daily_log_time", .text)
            }
        }
        migrator.registerMigration("v1_seed") { db in
            try Seed.apply(db)
        }
        return migrator
    }
}
```

```swift
import Foundation
import GRDB

enum Seed {
    static func apply(_ db: Database) throws {
        // 糖度（含分糖檔位）
        let levels: [SugarLevel] = [
            .init(code: "none",  label: "無糖",   multiplier: 0.0, sortOrder: 0),
            .init(code: "s10",   label: "一分糖", multiplier: 0.1, sortOrder: 1),
            .init(code: "s30",   label: "微糖",   multiplier: 0.3, sortOrder: 2),
            .init(code: "half",  label: "半糖",   multiplier: 0.5, sortOrder: 3),
            .init(code: "s70",   label: "少糖",   multiplier: 0.7, sortOrder: 4),
            .init(code: "s80",   label: "八分糖", multiplier: 0.8, sortOrder: 5),
            .init(code: "full",  label: "全糖",   multiplier: 1.0, sortOrder: 6),
        ]
        for l in levels { try l.insert(db) }

        // 容量
        for s in [SizeOption(code: "m", label: "中杯", ml: 500),
                  SizeOption(code: "l", label: "大杯", ml: 700),
                  SizeOption(code: "xl", label: "特大杯", ml: 1000)] {
            try s.insert(db)
        }

        // 加料（單份糖分/熱量，皆為參考估算）
        let toppings: [Topping] = [
            .init(id: nil, name: "珍珠",   sugarG: 15, kcal: 100, isEstimated: true, sortOrder: 0),
            .init(id: nil, name: "椰果",   sugarG: 8,  kcal: 40,  isEstimated: true, sortOrder: 1),

        for var t in toppings { try t.insert(db) }

        // 示範店家與品項（全糖、base_size_ml=700 基準；數值為參考估算，待資料策展擴充）
        var shop = DrinkShop(id: nil, name: "示範飲料店", sortOrder: 0)

                  baseSugarG: 45, baseKcal: 190, baseSizeML: 700, isEstimated: true),
            .init(id: nil, shopId: shopId, name: "多多綠", category: "果茶",
                  baseSugarG: 55, baseKcal: 240, baseSizeML: 700, isEstimated: true),

        for var it in items { try it.insert(db) }

        // 使用者設定單列
        try UserSetting(id: 1, dailySugarTargetG: 50,
                        notifyThreshold: true, notifyDailyLog: false,
                        notifyDailyLogTime: nil).insert(db)
    }
}
```

`DrinkSugar/Core/Logic/SugarCalculator.swift`：

```swift
import Foundation

enum SugarCalculator {
    static let kcalPerSugarGram = 4.0

    static func drinkSugar(baseSugarG: Double, multiplier: Double, sizeML: Int, baseSizeML: Int) -> Double {
        let ratio = Double(sizeML) / Double(baseSizeML)
        return baseSugarG * multiplier * ratio
    }

    static func drinkKcal(baseKcal: Double, baseSugarG: Double, multiplier: Double, sizeML: Int, baseSizeML: Int) -> Double {
        let ratio = Double(sizeML) / Double(baseSizeML)
        return (baseKcal - baseSugarG * (1 - multiplier) * kcalPerSugarGram) * ratio
    }

    static func toppingsSugar(_ toppings: [(sugarG: Double, qty: Int)]) -> Double {
        toppings.reduce(0) { $0 + $1.sugarG * Double($1.qty) }
    }

    static func toppingsKcal(_ toppings: [(kcal: Double, qty: Int)]) -> Double {
        toppings.reduce(0) { $0 + $1.kcal * Double($1.qty) }
    }

    static func totalSugar(baseSugarG: Double, multiplier: Double, sizeML: Int, baseSizeML: Int,
                           toppings: [(sugarG: Double, qty: Int)]) -> Double {
        drinkSugar(baseSugarG: baseSugarG, multiplier: multiplier, sizeML: sizeML, baseSizeML: baseSizeML)
            + toppingsSugar(toppings)
    }

    static func totalKcal(baseKcal: Double, baseSugarG: Double, multiplier: Double, sizeML: Int, baseSizeML: Int,
                          toppings: [(kcal: Double, qty: Int)]) -> Double {
        drinkKcal(baseKcal: baseKcal, baseSugarG: baseSugarG, multiplier: multiplier, sizeML: sizeML, baseSizeML: baseSizeML)
            + toppingsKcal(toppings)
    }
}
```

```swift
import Foundation
import GRDB

struct DailyTotal: Equatable {
    let day: Date
    let sugarG: Double
    let kcal: Double
}

protocol IntakeStoring {
    @discardableResult
    func save(_ log: IntakeLog, toppings: [IntakeLogTopping]) throws -> Int64
    func logs(on day: Date) throws -> [IntakeLog]
    func dailyTotalSugar(on day: Date) throws -> Double
    func delete(logId: Int64) throws
    func dailyTotals(from: Date, to: Date) throws -> [DailyTotal]
}

struct IntakeRepository: IntakeStoring {
    let db: AppDatabase
    init(_ db: AppDatabase) { self.db = db }

    private func dayBounds(_ day: Date) -> (Date, Date) {
        let cal = Calendar.current
        let start = cal.startOfDay(for: day)
        let end = cal.date(byAdding: .day, value: 1, to: start)!
        return (start, end)
    }

    @discardableResult
    func save(_ log: IntakeLog, toppings: [IntakeLogTopping]) throws -> Int64 {
        try db.dbQueue.write { d in
            var l = log
            try l.insert(d)
            let logId = l.id!
            for var t in toppings {
                t.intakeLogId = logId
                try t.insert(d)
            }
            return logId
        }
    }

    func logs(on day: Date) throws -> [IntakeLog] {
        let (start, end) = dayBounds(day)
        return try db.dbQueue.read {
            try IntakeLog
                .filter(Column("logged_at") >= start && Column("logged_at") < end)
                .order(Column("logged_at").desc)
                .fetchAll($0)
        }
    }

    func dailyTotalSugar(on day: Date) throws -> Double {
        let (start, end) = dayBounds(day)
        return try db.dbQueue.read { d in
            try Double.fetchOne(d, sql: """
                SELECT COALESCE(SUM(sugar_g), 0) FROM intake_log
                WHERE logged_at >= ? AND logged_at < ?
                """, arguments: [start, end]) ?? 0
        }
    }

    func delete(logId: Int64) throws {
        _ = try db.dbQueue.write { d in
            try IntakeLog.deleteOne(d, key: logId)
        }
    }

    func dailyTotals(from: Date, to: Date) throws -> [DailyTotal] {
        let cal = Calendar.current
        let start = cal.startOfDay(for: from)
        let end = cal.date(byAdding: .day, value: 1, to: cal.startOfDay(for: to))!
        let rows = try db.dbQueue.read { d in
            try Row.fetchAll(d, sql: """
                SELECT date(logged_at, 'localtime') AS day,
                       COALESCE(SUM(sugar_g),0) AS sugar,
                       COALESCE(SUM(kcal),0) AS kcal
                FROM intake_log
                WHERE logged_at >= ? AND logged_at < ?
                GROUP BY day ORDER BY day
                """, arguments: [start, end])
        }
        let df = DateFormatter(); df.dateFormat = "yyyy-MM-dd"; df.timeZone = .current
        return rows.compactMap { row in
            guard let dayStr: String = row["day"], let day = df.date(from: dayStr) else { return nil }
            return DailyTotal(day: day, sugarG: row["sugar"], kcal: row["kcal"])
        }
    }
}
```

```swift
import Foundation

struct AdviceInput {
    let todaySugarG: Double
    let targetG: Double
    let consecutiveOnTargetDays: Int
}

enum AdviceRule {
    static func message(for input: AdviceInput) -> String {
        if input.todaySugarG > input.targetG {
            return "今天糖分超標囉，明天可以試試改點半糖或無糖。"
        }
        if input.targetG > 0, input.todaySugarG >= input.targetG * 0.8 {
            return "已接近今日糖分上限，下一杯建議選微糖或無糖。"
        }
        if input.consecutiveOnTargetDays >= 3 {
            return "連續 \(input.consecutiveOnTargetDays) 天達標，控糖做得很好，繼續保持！"
        }
        return "今天的糖分還在額度內，喝得開心也記得適量。"
    }
}
```

```swift
        }
    }

    func delete(_ log: IntakeLog) {
        guard let id = log.id else { return }
        try? intake.delete(logId: id)
        reload()
    }
}
```

```swift
import Foundation

@MainActor
final class AddDrinkViewModel: ObservableObject {
    @Published var shops: [DrinkShop] = []
    @Published var items: [DrinkItem] = []
    @Published var levels: [SugarLevel] = []
    @Published var sizes: [SizeOption] = []
    @Published var toppings: [Topping] = []

    @Published var selectedShop: DrinkShop?
    @Published var selectedItem: DrinkItem?
    @Published var selectedLevel: SugarLevel?
    @Published var selectedSize: SizeOption?
    @Published var selectedToppingQty: [Int64: Int] = [:]

    @Published var isManual: Bool = false
    @Published var customName: String = ""
    @Published var customSugarText: String = ""
    @Published var customKcalText: String = ""

    @Published var previewSugar: Double = 0
    @Published var previewKcal: Double = 0

    private let catalog: CatalogReading
    private let intake: IntakeStoring

    init(catalog: CatalogReading, intake: IntakeStoring) {
        self.catalog = catalog
        self.intake = intake
    }

    func load() {
        levels = (try? catalog.sugarLevels()) ?? []
        sizes = (try? catalog.sizes()) ?? []
        toppings = (try? catalog.toppings()) ?? []
        shops = (try? catalog.shops()) ?? []
        selectedShop = shops.first
        if let s = selectedShop { selectShop(s) }
        selectedLevel = levels.first { $0.code == "full" } ?? levels.last
        selectedSize = sizes.first { $0.code == "l" } ?? sizes.first
    }

    func selectShop(_ shop: DrinkShop) {
        selectedShop = shop
        items = (try? catalog.items(shopId: shop.id ?? -1)) ?? []
        selectedItem = items.first
    }

    private func toppingSugarPairs() -> [(sugarG: Double, qty: Int)] {
        toppings.compactMap { t in
            guard let id = t.id, let q = selectedToppingQty[id], q > 0 else { return nil }
            return (t.sugarG, q)
        }
    }
    private func toppingKcalPairs() -> [(kcal: Double, qty: Int)] {
        toppings.compactMap { t in
            guard let id = t.id, let q = selectedToppingQty[id], q > 0 else { return nil }
            return ((t.kcal ?? 0), q)
        }
    }

    func recomputePreview() {
        if isManual {
            previewSugar = Double(customSugarText) ?? 0
            previewKcal = Double(customKcalText) ?? 0
            return
        }
        guard let item = selectedItem, let level = selectedLevel, let size = selectedSize else {
            previewSugar = 0; previewKcal = 0; return
        }
        previewSugar = SugarCalculator.totalSugar(
            baseSugarG: item.baseSugarG, multiplier: level.multiplier,
            sizeML: size.ml, baseSizeML: item.baseSizeML, toppings: toppingSugarPairs())
        previewKcal = SugarCalculator.totalKcal(
            baseKcal: item.baseKcal, baseSugarG: item.baseSugarG, multiplier: level.multiplier,
            sizeML: size.ml, baseSizeML: item.baseSizeML, toppings: toppingKcalPairs())
    }

    @discardableResult
    func save() -> Bool {
        recomputePreview()
        let log: IntakeLog
        if isManual {
            guard !customName.trimmingCharacters(in: .whitespaces).isEmpty else { return false }
            log = IntakeLog(id: nil, loggedAt: Date(), itemId: nil, customName: customName,
                            sugarG: previewSugar, kcal: previewKcal > 0 ? previewKcal : nil,
                            sugarLevelCode: nil, sizeCode: nil, note: nil)
        } else {
            guard let item = selectedItem else { return false }
            log = IntakeLog(id: nil, loggedAt: Date(), itemId: item.id, customName: item.name,
                            sugarG: previewSugar, kcal: previewKcal,
                            sugarLevelCode: selectedLevel?.code, sizeCode: selectedSize?.code, note: nil)
        }
        let toppingRows: [IntakeLogTopping] = toppings.compactMap { t in
            guard let id = t.id, let q = selectedToppingQty[id], q > 0 else { return nil }
            return IntakeLogTopping(id: nil, intakeLogId: 0, toppingId: id, customName: t.name,
                                    qty: q, sugarG: t.sugarG * Double(q),
                                    kcal: (t.kcal ?? 0) * Double(q))
        }
        do { try intake.save(log, toppings: toppingRows); return true }
        catch { return false }
    }
}
```

```swift
import Foundation

enum HistoryRange { case week, month }

@MainActor
final class HistoryViewModel: ObservableObject {
    @Published var range: HistoryRange = .week
    @Published var points: [DailyTotal] = []
    @Published var overTargetDays: Int = 0
    @Published var targetG: Double = 50

    private let intake: IntakeStoring
    private let setting: SettingStoring

    init(intake: IntakeStoring, setting: SettingStoring) {
        self.intake = intake
        self.setting = setting
    }

    func setRange(_ r: HistoryRange) { range = r; load() }

    func load() {
        let cal = Calendar.current
        let to = Date()
        let days = (range == .week) ? -6 : -29
        let from = cal.date(byAdding: .day, value: days, to: to)!
        targetG = (try? setting.setting().dailySugarTargetG) ?? 50
        points = (try? intake.dailyTotals(from: from, to: to)) ?? []
        overTargetDays = points.filter { $0.sugarG > targetG }.count
    }
}
```

```swift
@MainActor
final class SettingsViewModel: ObservableObject {
    @Published var targetG: Double = 50
    @Published var notifyThreshold: Bool = true
    @Published var notifyDailyLog: Bool = false
    @Published var dailyLogTime: Date = SettingsViewModel.defaultTime()

    private let setting: SettingStoring
    init(setting: SettingStoring) { self.setting = setting }

    static func defaultTime() -> Date {
        Calendar.current.date(from: DateComponents(hour: 21, minute: 0)) ?? Date()
    }

    func load() {
        guard let s = try? setting.setting() else { return }
        targetG = s.dailySugarTargetG
        notifyThreshold = s.notifyThreshold
        notifyDailyLog = s.notifyDailyLog

            let df = DateFormatter(); df.dateFormat = "HH:mm"
            if let t = df.date(from: str) { dailyLogTime = t }
        }
    }

    func save() {
        let df = DateFormatter(); df.dateFormat = "HH:mm"
        let s = UserSetting(id: 1, dailySugarTargetG: targetG,
                            notifyThreshold: notifyThreshold, notifyDailyLog: notifyDailyLog,
                            notifyDailyLogTime: df.string(from: dailyLogTime))
        try? setting.update(s)
    }
}
```

```swift
import Foundation
import UserNotifications

enum NotificationDecision {
    static func shouldWarn(total: Double, target: Double, thresholdRatio: Double = 0.8) -> Bool {
        guard target > 0 else { return false }
        return total >= target * thresholdRatio
    }
}

struct NotificationScheduler {
    private let center = UNUserNotificationCenter.current()
    private let thresholdId = "threshold_warning"
    private let dailyLogId = "daily_log_reminder"

    func requestAuthorization() async -> Bool {
        (try? await center.requestAuthorization(options: [.alert, .sound, .badge])) ?? false
    }

    func scheduleThresholdWarningIfNeeded(total: Double, target: Double, enabled: Bool) {
        center.removePendingNotificationRequests(withIdentifiers: [thresholdId])
        guard enabled, NotificationDecision.shouldWarn(total: total, target: target) else { return }
        let content = UNMutableNotificationContent()
        content.title = "糖分提醒"
        content.body = total > target
            ? "今天糖分已超標，下一杯建議選無糖。"
            : "今天糖分已接近上限，注意一下囉。"
        let trigger = UNTimeIntervalNotificationTrigger(timeInterval: 1, repeats: false)
        center.add(UNNotificationRequest(identifier: thresholdId, content: content, trigger: trigger))
    }

    func scheduleDailyLogReminder(enabled: Bool, hour: Int, minute: Int) {
        center.removePendingNotificationRequests(withIdentifiers: [dailyLogId])
        guard enabled else { return }
        let content = UNMutableNotificationContent()
        content.title = "別忘了記錄"
        content.body = "今天喝了什麼飲料？花 10 秒記一下吧。"
        var comps = DateComponents(); comps.hour = hour; comps.minute = minute
        let trigger = UNCalendarNotificationTrigger(dateMatching: comps, repeats: true)
        center.add(UNNotificationRequest(identifier: dailyLogId, content: content, trigger: trigger))
    }

    func cancelAll() {
        center.removeAllPendingNotificationRequests()
    }
}
```

```swift
import SwiftUI

struct RootView: View {
    let env: AppEnvironment
    @State private var showAdd = false
    @StateObject private var todayVM: TodayViewModel

    init(env: AppEnvironment) {
        self.env = env
        _todayVM = StateObject(wrappedValue: TodayViewModel(intake: env.intake, setting: env.setting))
    }

    var body: some View {
        TabView {
            TodayView(viewModel: todayVM, onAddTapped: { showAdd = true })
                .tabItem { Label("今日", systemImage: "cup.and.saucer.fill") }
            HistoryView(viewModel: HistoryViewModel(intake: env.intake, setting: env.setting))
                .tabItem { Label("歷史", systemImage: "chart.bar.fill") }
            SettingsView(viewModel: SettingsViewModel(setting: env.setting))
                .tabItem { Label("設定", systemImage: "gearshape.fill") }
        }
        .sheet(isPresented: $showAdd, onDismiss: { todayVM.reload(); scheduleThreshold() }) {
            AddDrinkView(viewModel: AddDrinkViewModel(catalog: env.catalog, intake: env.intake))
        }
        .task {
            _ = await env.scheduler.requestAuthorization()
            todayVM.reload()
            scheduleThreshold()
            scheduleDailyLog()
        }
    }

    // MARK: - 通知接線（Task 13 的排程器在此使用，ViewModel 不需改動）
    private func scheduleThreshold() {
        let enabled = (try? env.setting.setting().notifyThreshold) ?? true
        env.scheduler.scheduleThresholdWarningIfNeeded(
            total: todayVM.todaySugar, target: todayVM.targetG, enabled: enabled)
    }

    private func scheduleDailyLog() {
        guard let s = try? env.setting.setting() else { return }
        let parts = (s.notifyDailyLogTime ?? "21:00").split(separator: ":").compactMap { Int($0) }
        let hour = parts.first ?? 21
        let minute = parts.count > 1 ? parts[1] : 0
        env.scheduler.scheduleDailyLogReminder(enabled: s.notifyDailyLog, hour: hour, minute: minute)
    }
}
```
